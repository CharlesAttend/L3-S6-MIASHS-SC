  for(i in 1:sample_lenght){
    tmp = c()

              abs((i-1)/n - sorted_unif[i])
            )
        )
    }
    ks_ech = c(ks_ech, max(tmp))
  }
  return(ks_ech)
}

            )
        )
    }
    lilliefors_ech = c(lilliefors_ech, max(tmp))
  }
  return(lilliefors_ech)
}
